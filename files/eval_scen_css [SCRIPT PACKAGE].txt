[script 1.0]

script package eval_scen_css
is

  function opt_mdl_id(
    i_doc_asset           mem_doc_asset
    ,i_calc_md_scen_id id table code_md_scen
    ,i_eval_date date
  ) return id table code_eval_mdl
  is
  begin
    if i_doc_asset.obj_extn.ass_type_id in (btt.asset.obj_extn.ass_type.class.cfc) then
      return code_eval_mdl.bn;
    else
      return null;
    end if;

  exception
    when others then
      err.raise_fa('eval_scen_css.opt_mdl_id('||i_doc_asset||', '||i_calc_md_scen_id||', '||i_eval_date||')');
  end opt_mdl_id;

end eval_scen_css;
